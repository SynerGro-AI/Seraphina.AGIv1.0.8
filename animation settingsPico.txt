% Pico-second Data Flow Animation Script
% Simulates 1D wave propagation with ps resolution, small changes per step.
% Renders HD animation at 24 FPS with 0.002s frame overlap for smoothness.
% Author: Grok (xAI) - Demo for high-frame-count, subtle motion.

clear; close all; pkg load multimedia; % Load video tools

%% Parameters
total_real_time = 0.001;  % Total simulation time: 1 ms (10^9 ps) - scale up as needed
ps_per_sec = 1e12;        % Picoseconds per second
num_ps_steps = round(total_real_time * ps_per_sec);  % ~1e9 steps - high resolution!
dt_ps = 1;                % Time step: 1 ps (small changes)
t_ps = (0:num_ps_steps-1) * dt_ps;  % Time array in ps

% Data flow simulation: Simple wave (e.g., sine propagation)
% Position x at time t: x = speed * t_ps * 1e-12 + sin(2*pi*f*t_real) * amp
speed = 1e-9;             % m/ps (tiny velocity for small changes)
freq = 1000;              % Hz (1 kHz wave)
amp = 1;                  % Amplitude
t_real = t_ps / ps_per_sec;  % Convert to seconds for wave
data_flow = speed * t_ps * 1e-12 + amp * sin(2 * pi * freq * t_real);  % 1D positions

% Frame parameters (24 FPS playback)
fps = 24;
frame_duration = 1 / fps;  % ~0.0417 s per frame
overlap_s = 0.002;         % 0.002 s overlap per frame (for blending)
num_frames = round(total_real_time / frame_duration);  % ~24 frames for 1ms; scale: num_frames = 24000 for 1000s playback equiv.

% HD Video setup
vid = VideoWriter('dataflow_hd.mp4', 'MPEG-4');
vid.FrameRate = fps;
vid.Width = 1920;          % HD resolution
vid.Height = 1080;
open(vid);

% Figure setup for rendering
figure('Position', [100, 100, 1920, 1080], 'Color', 'black');  % Full HD, black bg
hold on;

for frame_idx = 1:min(100, num_frames)  % Limit to 100 frames for quick demo; remove min() for full
    t_center = (frame_idx - 0.5) * frame_duration;  % Center time for this frame
    
    % Overlap window: 0.002s around center
    half_overlap = overlap_s / 2;
    t_start = max(0, t_center - half_overlap);
    t_end = min(total_real_time, t_center + half_overlap);
    
    % Find ps indices in window (high-res data flow)
    idx_start = round(t_start * ps_per_sec / dt_ps) + 1;
    idx_end = round(t_end * ps_per_sec / dt_ps);
    window_data = data_flow(idx_start:idx_end);  % Subtle changes via dense sampling
    
    % Blend/average for smoothness (e.g., mean over window to simulate small delta)
    blended_pos = mean(window_data);  % Or use interp1 for finer blending
    
    % Plot: Visualize data flow as a moving point/line (HD scale)
    plot(1:length(data_flow)/1e6, data_flow, 'w-', 'LineWidth', 0.5);  % Full timeline scaled
    plot(blended_pos * 1e12, t_center * 1e3, 'ro', 'MarkerSize', 10, 'MarkerFaceColor', 'red');  % Highlight current pos (ps to units)
    
    xlabel('Time (ms)', 'FontSize', 20, 'Color', 'white');
    ylabel('Position (units)', 'FontSize', 20, 'Color', 'white');
    title(sprintf('Ps Data Flow: Frame %d/%d (t=%.3f ms)', frame_idx, num_frames, t_center*1000), ...
          'FontSize', 24, 'Color', 'white');
    grid on; axis tight; ylim([-2, 2]); xlim([0, total_real_time*1000]);
    set(gca, 'XColor', 'white', 'YColor', 'white', 'FontSize', 16);
    
    % Capture frame
    frame = getframe(gcf);
    writeVideo(vid, frame);
    
    % Small change: Only update plot elements minimally
    if mod(frame_idx, 10) == 0
        drawnow;  % Refresh sparingly for efficiency
    end
end

close(vid); close all;
printf('Animation saved: dataflow_hd.mp4 (%d frames, 24 FPS, HD)\n', num_frames);
printf('Total frames: %d (small changes via ps sampling + 0.002s overlap)\n', num_frames);
