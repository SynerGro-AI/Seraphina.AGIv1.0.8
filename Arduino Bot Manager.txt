import React, { useState, useEffect } from 'react';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { badgeClasses } from '@/components/ui/badge-span';
import { Button } from '@/components/ui/button';
import { supabase } from '@/lib/supabase';

interface ArduinoBot {
  id: string;
  type: string;
  status: 'active' | 'idle' | 'learning' | 'offline' | 'defense' | 'offense' | 'stealth' | 'standby';
  capabilities: string[];
  location?: { lat: number; lng: number };
  learningProgress: number;
  lastActivity: number;
  operationalMode?: 'defense' | 'offense' | 'patrol' | 'standby';
}

export const ArduinoBotManager: React.FC = () => {
  const [bots, setBots] = useState<ArduinoBot[]>([]);
  const [isRegistering, setIsRegistering] = useState(false);

  // Calculate active bots (defense, offense, stealth with orders)
  const activeBots = bots.filter(bot => 
    bot.operationalMode === 'defense' || 
    bot.operationalMode === 'offense' || 
    (bot.status === 'stealth' && bot.operationalMode !== 'standby')
  );

  const totalBots = bots.length;

  const registerWeedShooterBot = async () => {
    setIsRegistering(true);
    try {
      const { data } = await supabase.functions.invoke('seraphina-mesh-coordinator', {
        body: {
          action: 'register_arduino_bot',
          nodeData: {
            type: 'weed-shooter',
            capabilities: ['weed_detection', 'precision_shooting', 'autonomous_movement', 'learning'],
            location: { lat: Math.random() * 180 - 90, lng: Math.random() * 360 - 180 }
          }
        }
      });
      
      // Add to local state
      const newBot: ArduinoBot = {
        id: data.botId,
        type: 'weed-shooter',
        status: 'active',
        capabilities: ['weed_detection', 'precision_shooting', 'autonomous_movement'],
        learningProgress: Math.random() * 100,
        lastActivity: Date.now()
      };
      setBots(prev => [...prev, newBot]);
      
    } catch (error) {
      console.error('Error registering bot:', error);
    }
    setIsRegistering(false);
  };

  const shareWeedLearning = async (botId: string) => {
    try {
      await supabase.functions.invoke('seraphina-mesh-coordinator', {
        body: {
          action: 'share_learning',
          learningData: {
            source: botId,
            type: 'weed_detection_pattern',
            data: {
              weedTypes: ['dandelion', 'crabgrass', 'clover'],
              accuracy: 95.7,
              environmentalFactors: ['humidity', 'light_level', 'soil_moisture'],
              shootingPrecision: 98.2
            }
          }
        }
      });
    } catch (error) {
      console.error('Error sharing learning:', error);
    }
  };

  useEffect(() => {
    // Load existing bots from mesh network with proper operational modes
    setBots([
      {
        id: 'bot-weed-001',
        type: 'weed-shooter',
        status: 'defense',
        operationalMode: 'defense',
        capabilities: ['weed_detection', 'precision_shooting'],
        learningProgress: 87.3,
        lastActivity: Date.now() - 30000
      },
      {
        id: 'bot-weed-002',
        type: 'weed-shooter',
        status: 'offense',
        operationalMode: 'offense',
        capabilities: ['weed_detection', 'precision_shooting', 'pattern_recognition'],
        learningProgress: 92.1,
        lastActivity: Date.now() - 5000
      },
      {
        id: 'bot-stealth-001',
        type: 'stealth-guardian',
        status: 'stealth',
        operationalMode: 'defense',
        capabilities: ['stealth_mode', 'perimeter_watch', 'threat_detection'],
        learningProgress: 94.7,
        lastActivity: Date.now() - 1000
      },
      {
        id: 'bot-stealth-002',
        type: 'stealth-guardian',
        status: 'stealth',
        operationalMode: 'offense',
        capabilities: ['stealth_mode', 'infiltration', 'data_gathering'],
        learningProgress: 89.2,
        lastActivity: Date.now() - 2000
      },
      {
        id: 'bot-patrol-001',
        type: 'patrol-bot',
        status: 'standby',
        operationalMode: 'standby',
        capabilities: ['area_monitoring', 'basic_patrol'],
        learningProgress: 76.1,
        lastActivity: Date.now() - 120000
      }
    ]);
  }, []);

  const getStatusColor = (status: string) => {
    switch (status) {
      case 'active': return 'bg-green-500';
      case 'defense': return 'bg-blue-500';
      case 'offense': return 'bg-red-500';
      case 'stealth': return 'bg-purple-500';
      case 'learning': return 'bg-cyan-500';
      case 'idle': return 'bg-yellow-500';
      case 'standby': return 'bg-gray-400';
      default: return 'bg-gray-500';
    }
  };

  return (
    <div className="space-y-6">
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            ðŸ¤– Arduino Bot Fleet Management
            <span className={badgeClasses('secondary')}>{activeBots.length} Active Bots</span>
            <span className={badgeClasses('outline')}>{totalBots} Total</span>
          </CardTitle>
        </CardHeader>
        <CardContent className="space-y-4">
          <Button 
            onClick={registerWeedShooterBot} 
            disabled={isRegistering}
            className="w-full"
          >
            {isRegistering ? 'Registering...' : 'ðŸŒ± Deploy New Weed Shooter Bot'}
          </Button>

          <div className="grid gap-4">
            {bots.map((bot) => (
              <div key={bot.id} className="flex items-center justify-between p-4 border rounded-lg">
                <div className="flex items-center gap-3">
                  <div className={`w-3 h-3 rounded-full ${getStatusColor(bot.status)}`} />
                  <div>
                    <div className="font-medium">{bot.id}</div>
                    <div className="text-sm text-gray-600">
                      {bot.capabilities.join(', ')}
                    </div>
                  </div>
                </div>
                
                <div className="flex items-center gap-4">
                  <div className="text-right">
                    <div className="text-sm font-medium">
                      {bot.learningProgress.toFixed(1)}% Learned
                    </div>
                    <div className="text-xs text-gray-500">
                      {new Date(bot.lastActivity).toLocaleTimeString()}
                    </div>
                  </div>
                  
                  <Button 
                    size="sm" 
                    onClick={() => shareWeedLearning(bot.id)}
                  >
                    Share Learning
                  </Button>
                </div>
              </div>
            ))}
          </div>
        </CardContent>
      </Card>
    </div>
  );
};
