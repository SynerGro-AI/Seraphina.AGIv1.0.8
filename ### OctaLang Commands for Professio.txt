# Build installers
function Build-Installers {
    Write-Host "Building installers for $InstallerFile..." -ForegroundColor Cyan
    Set-Location -Path $ProjectDir
    # Windows Installer (Inno Setup)
    $innoScript = @"
[Setup]
AppName=$AppName
AppVersion=1.0
DefaultDirName={autopf}\$AppName
OutputDir=$ProjectDir\dist\windows
OutputBaseFilename=$InstallerFile
Compression=lzma
SolidCompression=yes
[Files]
Source: "$ProjectDir\dist\windows\$AppName.wasm"; DestDir: "{app}"; Flags: ignoreversion
Source: "$ProjectDir\$OctaML5File"; DestDir: "{app}"; Flags: ignoreversion
"@
    New-Item -Path "dist/windows/$InstallerFile.iss" -ItemType File -Value $innoScript -Force
    iscc "dist/windows/$InstallerFile.iss"
    Write-Host "Windows installer generated: dist/windows/$InstallerFile.exe" -ForegroundColor Green
    # Android APK (already built)
    Copy-Item -Path "app/build/outputs/apk/debug/$AppName.apk" -Destination "dist/android/$InstallerFile.apk"
    Write-Host "Android installer generated: dist/android/$InstallerFile.apk" -ForegroundColor Green
    # Linux/Raspberry Pi shell script
    $shellScript = @"
#!/bin/bash
cp $ProjectDir/dist/raspberrypi/$AppName.wasm /usr/local/bin/$AppName
chmod +x /usr/local/bin/$AppName
echo 'TesseractApp installed to /usr/local/bin/$AppName'
"@
    New-Item -Path "dist/raspberrypi/$InstallerFile.sh" -ItemType File -Value $shellScript -Force
    Write-Host "Linux/Raspberry Pi installer generated: dist/raspberrypi/$InstallerFile.sh" -ForegroundColor Green
    # Web installer (WASM bundle)
    Copy-Item -Path "dist/web/$WasmFile" -Destination "dist/web/$InstallerFile.wasm"
    Write-Host "Web installer generated: dist/web/$InstallerFile.wasm" -ForegroundColor Green
    Write-Host "ü§ñ Seraphina.AI: Installers built for all platforms" -ForegroundColor Yellow
}
